library(igraph)
g <- random.graph.game(100, 1/7, directed=TRUE)
plot.igraph(g)
betweenness(g)
degree(g)
closeness(g)
centrality (g)
evcent(g)

###measuring clustering coefficient and path length
transitivity(g)
average.path.length(g)
hist(degree(g))
